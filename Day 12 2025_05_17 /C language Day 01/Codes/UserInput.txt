//Take input from users

#include<stdio.h>
int main(){

//For int

int age;
printf("Enter your age:");
scanf("%d",&age);

printf("\nAge:%d\n",age);


//For Double

double value;
printf("\nEnter your value:");
scanf("%lf",&value);

printf("Value:%lf\n",value);


//For Char
	char letter;
	printf("\nEnter the Letter:");
	scanf("%c",&letter); 
	
	printf("Letter:%c",letter);

}
